    def _parse_preview_ports(self, res):
        # res = '[(a, b, c), (a, b, c)*]'
        x = ast.literal_eval(res)
        preview_ports = []
        for tupl in x:
            preview_ports.append(int(tupl[0]))
        return preview_ports
