[docs]    def run_process(self):
        cmd = self.PATH
        # cmd = ''
        cmd += """gst-switch-srv \
                    --video-input-port=%s \
                    --audio-input-port=%s \
                    --control-port=%s \
                    --record=%s """ % (self.VIDEO_PORT, self.AUDIO_PORT, self.CONTROL_PORT, self.RECORD_FILE)
        proc = self.start_process(cmd)
        print "process:", proc
        if proc is None:
            print 'ERROR: Server unable to create process'
            pass
        else:
            print 'Created process with PID:%s', str(proc.pid)
            return proc



[docs]    def start_process(self, cmd):
        print 'Creating process %s' % (cmd)
        with open(os.devnull, 'w') as tempf:
            process = subprocess.Popen(cmd.split(), stdout=tempf, stderr=tempf,  bufsize=-1, shell=False)
        return process



[docs]    def terminate(self):
        """Stops the server
        Returns:
            True on success
            False on failure
        Parameters:
            None
        """
        print 'Killing server'
        proc = self.proc
        ret = True
        try:
            proc.terminate()
            print 'Server Killed'
        except:
            print 'Error killing server'
            ret = False
        return ret
